
CREATE TABLE PROJECTCAMPAIGN
( 
campaign_id NUMBER,
title VARCHAR2(255), 
start_date DATE NOT NULL, 
end_date DATE NOT NULL, 
funding_goal DECIMAL(19,4) NOT NULL, 
funding_received DECIMAL(19,4) DEFAULT 0, 
description VARCHAR2(255) NOT NULL, 
CONSTRAINT campaign_pk PRIMARY KEY (campaign_id)
);


CREATE TABLE PROJECTUSER
( 
user_id NUMBER,
first_name VARCHAR2(255) NOT NULL, 
last_name VARCHAR2(255) NOT NULL, 
password_hash VARCHAR2(255) NOT NULL,  
role VARCAHR2(20) CHECK(role IN(('user','admin'))
);

CREATE TABLE PROJECTTRANSACTION
( 
transaction_id NUMBER,
campaign_id NUMBER, 
user_id NUMBER, 
authorization_num VARCHAR2(255) NOT NULL,
payment_type VARCHAR2(255) NOT NULL,
payment_date DATE NOT NULL, 
amount DECIMAL(19,4) NOT NULL, CHECK(amount>0),
CONSTRAINT fk_campaign FOREIGN KEY (campaign_id) REFERENCES ProjectCampaign(campaign_id),
CONSTRAINT fk_user FOREIGN KEY (user_id) REFERENCES ProjectUser(user_id),
CONSTRAINT pk_transaction PRIMARY KEY (transaction_id)
);
